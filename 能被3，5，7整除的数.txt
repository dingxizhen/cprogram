#include<iostream>
using namespace std;

{
	int n,a1,a2,t;

	{
		if(n%3==0&&n%5==0&&n%7==0)
		{
			cout<<"3"<<" "<<"5"<<" "<<"7"<<endl;
		}
		if(n%3==0&&n%5==0&&n%7!=0)
		{
			cout<<"3"<<" "<<"5"<<endl;
		}
		if(n%3==0&&n%7==0&&n%5!=0)
				cout<<"3"<<" "<<"7"<<endl;
		if(n%5==0&&n%7==0&&n%3!=0)
				cout<<"5"<<" "<<"7"<<endl;
		if(n%3==0&&n%5!=0&&n%7!=0)
			cout<<"3"<<endl;
		if(n%5==0&&n%3!=0&&n%7!=0)
			cout<<"5"<<endl;

			cout<<"7"<<endl;
		if(n%3!=0&&n%5!=0&&n%7!=0)
			cout<<"n"<<endl;
	}
		return 0;
}
